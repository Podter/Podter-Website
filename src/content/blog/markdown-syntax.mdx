```ts
async function copy() {
  try {
    setCopyState("copying");
    await navigator.clipboard.writeText(code);
    setCopyState("copied");
  } catch (e) {
    console.error(e);
    setCopyState("error");
  } finally {
    setTimeout(() => {
      setCopyState(null);
    }, 1000);
  }
}
```
